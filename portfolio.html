#include 

using namespace std;

int main()
{
	// we will ask user to input any two value.
	int h = 0;
	int l = 0;

	cout << "Please enter two values:" << endl;
	cin >> h;
	cin >> l;

	cout << h << " + " << l << " = " << h + l << endl;
	cout << h << " - " << l << " = " << h - l << endl;
	cout << h << " * " << l << " = " << h * l << endl;
	cout << h << " / " << l << " = " << h / l << endl;
	cout << h << " % " << l << " = " << h % l << endl;
	return 0;
}
